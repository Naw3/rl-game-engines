# =============================================================================
# Python dependencies for the Connect4 AlphaZero pipeline.
#
# This file only covers Python-level deps. For the FULL setup (CUDA toolkit
# install + cuDNN wheel + Rust toolchain), run from the project root:
#
#   powershell -ExecutionPolicy Bypass -File .\setup.ps1
#
# After setup.ps1 has run once, this file alone can refresh the Python env:
#
#   pip install -r requirements.txt
#
# Why the index magic:
#   * `--index-url` replaces pip's default PyPI with the PyTorch CUDA wheel
#     server, so we get the CUDA build of `torch` (PyPI's torch is CPU-only).
#   * `--extra-index-url https://pypi.org/simple` re-adds PyPI as a fallback
#     so non-PyTorch packages (numpy, pygame, onnx, nvidia-cudnn-cu12, ...)
#     can still be found. Without this line, pip would only search the cu124
#     index and pygame etc. would fail with "No matching distribution found".
#
# To switch to CPU-only torch (no CUDA), comment out the two index lines and
# re-install. Everything else works either way.
# =============================================================================
--index-url https://download.pytorch.org/whl/cu124
--extra-index-url https://pypi.org/simple

# --- Core ---
torch                # CUDA 12.4 build (Connect 4 doesn't need torchvision — no images)
numpy>=1.24
pygame>=2.5
onnx>=1.15           # Model export to ONNX, consumed by tract-onnx in Rust.
onnxscript           # Required by torch.onnx.export in PyTorch 2.13+ (newer ONNX exporter).

# --- GPU runtime ---
onnxruntime-gpu      # Optional Python-side inference check / debugging.
nvidia-cudnn-cu12    # cuDNN 9.x for CUDA 12 — data-only wheel, ships DLLs in
                     # <site-packages>/nvidia/cudnn/bin/ but no Python module.
                     # Note: the cuDNN DLLs need a CUDA 12 toolkit on PATH
                     # (cublasLt64_12.dll etc.) — handled by setup.ps1 + the
                     # bench script. nvidia-cublas-cu12 and nvidia-cuda-nvrtc-cu12
                     # come in as transitive deps.
